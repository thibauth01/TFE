//youngr

Table account as acc {
  id int [pk, increment] // auto-increment
  first_name varchar
  last_name varchar
  email varchar
  username varchar
  password varchar
  birth_date date
  street varchar
  postcode int
  city varchar
  country varchar
  premium boolean
  
}

Table requester{
  id int [pk,increment]
  id_account int [ref: - acc.id]
  premium boolean [default:false]
}

Table worker {
  id int [pk,increment]
  maximum_distance int
  id_account int [ref: - acc.id]
  star double
}

Table woker_Typer_work{
  id int [pk,increment]
  id_worker int [ref: > worker.id]
  id_type_work int [ref: > type_work.id]
}

Table type_work{
  id int [pk, increment]
  name varchar 
}

Table availability{
  id int [pk,increment]
  id_worker int [ref: > worker.id]
  id_day int [ref: > day.id]
  
}

Table day{
  id int [pk,increment]
  nom varchar
}


Table work{
  id int [pk, increment]
  title varchar
  id_type int [ref: - type_work.id]
  description text
  id_requester int [ref: > requester.id]
  id_worker int [ref: > worker.id,default: null]
  min_age_worker int
  date_start date
  date_end date
  time_start date
  time_end date
  place varchar
  id_statut_progress varchar 
  paid boolean [default:false]
  cancelled boolean [default:false]
}


Table message{
  id int [pk,increment]
  id_sender int [ref: > account.id]
  id_receiver int [ref: > account.id]
  content varchar
  sendtime timestamp
}

Table notification{
  id int [pk,increment]
  id_sender int [ref: > account.id]
  id_receiver int [ref: > account.id]
  content varchar
  sendtime timestamp
}
